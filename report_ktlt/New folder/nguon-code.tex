void seidel_iterative_method (double a[][MAX], double result[][MAX], int n, int k)
{
    for (int i = 1; i <= n; i++)
        result[i][0] = a[i][n+1];

    for (int h = 1; h <= k; h++) {
        result[1][h] = result[1][0];
        for (int j = 1; j <= n; j++)
            result[1][h] += a[1][j] * result[j][h-1];
        for (int i = 2; i <= n; i++) {
            result[i][h] = result[i][0];
            for (int j = 1; j <= i-1; j++)
                result[i][h] += a[i][j] * result[j][h];
            for (int t = i; t <= n; t++)
                result[i][h] += a[i][t] * result[t][h-1];
        }
    }
}
\end{lstlisting}
